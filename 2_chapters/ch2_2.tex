\section{Ouverts et fermés}

	\subsection{Intérieurs}
	
		Soit $A\subset E$ et $\alpha\in E$
		
		\definition{Soit un point $\alpha \in A$ on dit que $\alpha$ est un point intérieur à $A$ si $B(\alpha,r) \subset A$ pour un certain réel $r>0$.
		}
		
		\definition{On appelle intérieur de $A$ l'ensemble des points intérieurs de $A$ 
			\[
				\mathring{A} = \{x\in E ~\vert ~x \text{ est intérieur à } A\}
			\] \vspace{-15pt}
		}
		
		\theorem{lem}{Soit $A \subset E$ alors $\mathring{A} \subset E$}{InteDansEVN}
		
		\theorem{lem}{Le passage à l'intérieur est une opération croissante pour l'inclusion.
		{\small (i.e. $A \subset B \Rightarrow \mathring{A} \subset \mathring{B}$)}}{CroissInte}
